## 插入项目源码链接 {#插入项目源码链接}

```elisp
(defun my/insert-ue-project-source-path ()
  (interactive)
  (let ((proj (read-string "Project: (Default - ShootThemUp)"))
        (type (read-string "Category: "))
        (file (read-string "File: "))
        (path)
        (attr))
    (if (string= "" proj)
        (setq proj "ShootThemUp"))
    (when (string= "" type)
      (insert (format "[[uep:%s][%s:%s]]" file proj file)))
    (unless (string= type "")
      (if (string-suffix-p ".h" file)
          (setq attr "Public")
        (setq attr "Private"))
      (setq path (format "%s/%s/%s" attr type file))
      (insert (format "[[uep:%s][%s:%s/%s]]" path proj type file)))))
(global-set-key (kbd "C-c u p") 'my/insert-ue-project-source-path)
```


## 打开源码文件 {#打开源码文件}

```elisp
(defun my/go-to-ue-project-source ()
  (interactive)
  (when (search-backward "[[uep:" nil t)
    (when (re-search-forward (rx "[[uep:" (group (0+ (not "]"))) "][" (group (0+ (not ":"))) ":" (group (0+ (not "]"))) "]]" ) nil t)
      (let ((file (string-join (mapcar #'string (match-string 1))))
            (proj (string-join (mapcar #'string (match-string 2)))))
        (org-open-file (format "%s/%s/Source/%s/%s" my/ue-project-path proj proj file))))))
(global-set-key (kbd "C-c u l") 'my/go-to-ue-project-source)
```
